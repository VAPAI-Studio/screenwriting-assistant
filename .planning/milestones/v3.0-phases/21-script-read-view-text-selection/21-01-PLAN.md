  2. Get `range = selection.getRangeAt(0)`. Check `containerRef.current.contains(range.commonAncestorContainer)` -- bail if selection is outside the script container.

     ```typescript
     function getSceneIdFromSelection(selection: Selection): string | null {
       const anchorNode = selection.anchorNode;
       if (!anchorNode) return null;
       const el = anchorNode.nodeType === Node.ELEMENT_NODE
         ? (anchorNode as Element)
         : anchorNode.parentElement;
       const sceneEl = el?.closest('[data-scene-id]');
       const id = sceneEl?.getAttribute('data-scene-id');
       return id || null;
     }
     ```
  7. Set `selectionState` with `{ text, lineCount, rect, sceneItemId }`.

**Add Shot handler:**
```typescript
const handleAddShotFromSelection = useCallback(() => {
  if (!selectionState) return;
  const { text, sceneItemId } = selectionState;

  const groupShots = (shots ?? []).filter(s =>
    sceneItemId === null
      ? s.scene_item_id === null
      : s.scene_item_id === sceneItemId
  );
  const maxNumber = groupShots.reduce((max, s) => Math.max(max, s.shot_number), 0);
  const maxSortOrder = groupShots.reduce((max, s) => Math.max(max, s.sort_order), -1);

  createMutation.mutate({
    scene_item_id: sceneItemId,
    shot_number: maxNumber + 1,
    script_text: text,
    sort_order: maxSortOrder + 1,
    source: 'user',
  });

  window.getSelection()?.removeAllRanges();
  setSelectionState(null);
}, [selectionState, shots, createMutation]);
```

const handleDismissSelection = useCallback(() => {
