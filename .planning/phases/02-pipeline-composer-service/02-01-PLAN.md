```python
"""
Pipeline Composer Service.
Analyzes user agents and maps them to template pipeline steps via AI.
"""
import hashlib

from uuid import UUID

from sqlalchemy.orm import Session

from ..config import settings
from ..models.database import Agent, AgentPipelineMap
from ..templates import get_template
from .ai_provider import chat_completion

logger = logging.getLogger(__name__)
```

**Class: PipelineComposer**

`__init__(self)`:
- `self._cache: Dict[str, List[Dict]] = {}` -- in-memory hash -> parsed AI result

`async def compose_pipeline(self, owner_id: UUID, db: Session) -> List[AgentPipelineMap]`:
1. Query all active agents for owner: `db.query(Agent).filter(Agent.owner_id == owner_id, Agent.is_active == True).all()`
2. If len(agents) == 0: delete existing mappings for owner, commit, return []
3. Compute cache key via `_compute_cache_key(agents)`
4. If cache hit: use cached result. If miss: call `_compose_batched(agents, targets)`
5. Store result in cache
6. Full-replace write: `db.query(AgentPipelineMap).filter(AgentPipelineMap.owner_id == owner_id).delete(synchronize_session='fetch')` then `db.flush()`
7. Create AgentPipelineMap instances for each entry, `db.add()` each, `db.commit()`
8. Return the created instances

`def _get_wizard_targets(self, template_id: str = "short_movie") -> List[Dict]`:
- Call `get_template(template_id)`
- Iterate phases -> subsections
- Filter: `"wizard" in sub.get("ui_pattern", "") and sub["key"] != "import_project"`
- Return list of `{"phase": phase["id"], "subsection_key": sub["key"], "name": sub["name"], "description": sub.get("description", "")}`

`def _compute_cache_key(self, agents: List[Agent]) -> str`:
- Sort agents by `str(a.id)` for determinism
- For each: `f"{agent.id}:{agent.system_prompt_template}:{agent.description or ''}:{agent.agent_type.value}"`
- Join with `"|"`, SHA-256 hash, return hexdigest

`async def _compose_batched(self, agents: List[Agent], targets: List[Dict]) -> List[Dict]`:
- Split agents into batches of `settings.PIPELINE_BATCH_SIZE`
- For each batch: `await self._call_ai_composition(batch, targets)`
- Concatenate all results

`async def _call_ai_composition(self, agents: List[Agent], targets: List[Dict]) -> List[Dict]`:
- Build system prompt with COMPOSITION_SYSTEM_PROMPT (containing target steps formatted)
- Build user prompt with COMPOSITION_USER_PROMPT (containing agent details)
- Call `await chat_completion(messages=[system, user], temperature=0, max_tokens=settings.PIPELINE_COMPOSITION_MAX_TOKENS, json_mode=True)`
- Parse JSON, extract "mappings" key
- Validate each mapping's agent_id against the set of actual agent IDs (discard unknowns)
- Convert agent_id strings to UUID
- Return list of valid mapping dicts

**Prompt constants** (at module level):
- Per user decision: include agent `system_prompt_template + description + agent_type` only
- Per user decision: include step `name + description` from template
- Do NOT include book titles, tags, or personality
- Per user decision: AI returns flat list of `{agent_id, phase, subsection_key, confidence, rationale}`
- See 02-RESEARCH.md "Composition Prompt Design" section for the exact prompt template

`def is_semantic_change(self, update_fields: Dict) -> bool`:
- `SEMANTIC_FIELDS = {"system_prompt_template", "description", "agent_type"}`
- Return `bool(set(update_fields.keys()) & SEMANTIC_FIELDS)`

Module-level singleton:

- Pitfall 5 (import_project false positive): explicit key exclusion in `_get_wizard_targets`
- Pitfall 6 (bulk delete + insert): use `synchronize_session='fetch'` and `db.flush()` after delete
