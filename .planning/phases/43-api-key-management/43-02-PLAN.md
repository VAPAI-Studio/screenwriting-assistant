2. **Add API methods to `frontend/src/lib/api.tsx`:**

   Add the import for `ApiKey`, `ApiKeyCreate`, `ApiKeyCreateResponse` from `'../types'` in the import statement at the top.

   Add these methods inside the `api` object, near the bottom (after the show/bible methods):

   ```typescript
   // API Keys (v5.0 -- Phase 43)
   createApiKey: async (data: ApiKeyCreate): Promise<ApiKeyCreateResponse> => {
     const response = await fetchWithTimeout(`${API_BASE_URL}/auth/api-keys`, {
       method: 'POST',
       headers: getHeaders(),
       body: JSON.stringify(data),
     });
     if (!response.ok) throw new Error('Failed to create API key');
     return response.json();
   },

   listApiKeys: async (): Promise<ApiKey[]> => {
     const response = await fetchWithTimeout(`${API_BASE_URL}/auth/api-keys`, {
       headers: getHeaders(),
     });
     if (!response.ok) throw new Error('Failed to list API keys');
     return response.json();
   },

   revokeApiKey: async (keyId: string): Promise<void> => {
     const response = await fetchWithTimeout(`${API_BASE_URL}/auth/api-keys/${keyId}`, {
       method: 'DELETE',
       headers: getHeaders(),
     });
     if (!response.ok) throw new Error('Failed to revoke API key');
   },
   ```
