  // Storyboard Frames (v3.2 -- Phase 29)

  async listStoryboardFrames(projectId: string, shotId: string): Promise<StoryboardFrame[]> {
    const response = await fetchWithTimeout(
      `${API_BASE_URL}/storyboard/${projectId}/shots/${shotId}/frames`,
      { headers: getHeaders() }
    );
    if (!response.ok) throw new Error('Failed to fetch storyboard frames');
    return response.json();
  },

  async uploadStoryboardFrame(projectId: string, shotId: string, formData: FormData): Promise<StoryboardFrame> {
    const response = await fetchWithTimeout(
      `${API_BASE_URL}/storyboard/${projectId}/shots/${shotId}/frames`,
      {
        method: 'POST',
        headers: { 'Authorization': getAuthToken() },
        body: formData,
      }
    );
    if (!response.ok) {
      const err = await response.json().catch(() => ({}));
      throw new Error(err.detail || 'Failed to upload storyboard frame');
    }
    return response.json();
  },

  async updateStoryboardFrame(projectId: string, frameId: string, data: { is_selected?: boolean }): Promise<StoryboardFrame> {
    const response = await fetchWithTimeout(
      `${API_BASE_URL}/storyboard/${projectId}/frames/${frameId}`,
      {
        method: 'PATCH',
        headers: getHeaders(),
        body: JSON.stringify(data),
      }
    );
    if (!response.ok) throw new Error('Failed to update storyboard frame');
    return response.json();
  },

  async deleteStoryboardFrame(projectId: string, frameId: string): Promise<void> {
    const response = await fetchWithTimeout(
      `${API_BASE_URL}/storyboard/${projectId}/frames/${frameId}`,
      {
        method: 'DELETE',
        headers: getHeaders(),
      }
    );
    if (!response.ok) throw new Error('Failed to delete storyboard frame');
  },

1. Rewrite `frontend/src/components/Layout/ModeToggle.tsx` to support three modes:

```typescript
import * as DropdownMenu from '@radix-ui/react-dropdown-menu';
import { useLocation, useNavigate, useMatch } from 'react-router-dom';
import { ChevronDown, PenLine, Clapperboard, Film } from 'lucide-react';
import { ROUTES } from '../../lib/constants';

type Mode = 'screenwriting' | 'breakdown' | 'storyboard';

const MODE_CONFIG: Record<Mode, { label: string; icon: typeof PenLine }> = {
  screenwriting: { label: 'Screenwriting', icon: PenLine },
  breakdown: { label: 'Script Breakdown', icon: Clapperboard },
  storyboard: { label: 'Storyboard', icon: Film },
};

export function ModeToggle() {
  const location = useLocation();
  const navigate = useNavigate();
  const projectMatch = useMatch('/projects/:projectId/*');
  const projectId = projectMatch?.params.projectId;

  if (!projectId) return null;

  const isBreakdown = location.pathname.endsWith('/breakdown');
  const isStoryboard = location.pathname.endsWith('/storyboard');
  const currentMode: Mode = isStoryboard ? 'storyboard'
    : isBreakdown ? 'breakdown'
    : 'screenwriting';
  const lastScreenwritingKey = `lastScreenwritingPath_${projectId}`;

  const handleSelect = (mode: Mode) => {
    if (mode === currentMode) return;

    // Save screenwriting path when leaving it
    if (currentMode === 'screenwriting') {
      localStorage.setItem(lastScreenwritingKey, location.pathname);
    }

    if (mode === 'breakdown') {
      navigate(ROUTES.PROJECT_BREAKDOWN(projectId));
    } else if (mode === 'storyboard') {
      navigate(ROUTES.PROJECT_STORYBOARD(projectId));
    } else {
      const last = localStorage.getItem(lastScreenwritingKey);
      navigate(last ?? ROUTES.PROJECT(projectId));
    }
  };

  const CurrentIcon = MODE_CONFIG[currentMode].icon;

  return (
    <DropdownMenu.Root>
      <DropdownMenu.Trigger asChild>
        <button
          className="flex items-center gap-1.5 px-3 py-1.5 text-sm font-medium rounded-md
            bg-muted/40 hover:bg-muted/70 text-foreground transition-colors"
        >
          <CurrentIcon className="h-3.5 w-3.5" />
          <span>{MODE_CONFIG[currentMode].label}</span>
          <ChevronDown className="h-3.5 w-3.5 text-muted-foreground" />
        </button>
      </DropdownMenu.Trigger>
      <DropdownMenu.Portal>
        <DropdownMenu.Content
          className="z-50 min-w-[190px] rounded-lg border border-border bg-card shadow-lg p-1"
          sideOffset={4}
        >
          {(Object.keys(MODE_CONFIG) as Mode[]).map((mode) => {
            const { label, icon: Icon } = MODE_CONFIG[mode];
            return (
              <DropdownMenu.Item
                key={mode}
                onSelect={() => handleSelect(mode)}
                className="flex items-center gap-2 px-3 py-2 text-sm rounded-md cursor-pointer
                  hover:bg-muted/60 text-foreground outline-none"
              >
                <Icon className="h-3.5 w-3.5" />
                <span>{label}</span>
                {currentMode === mode && (
                  <span className="ml-auto text-xs" style={{ color: 'hsl(var(--accent))' }}>Active</span>
                )}
              </DropdownMenu.Item>
            );
          })}
        </DropdownMenu.Content>
      </DropdownMenu.Portal>
    </DropdownMenu.Root>
  );
}
```
