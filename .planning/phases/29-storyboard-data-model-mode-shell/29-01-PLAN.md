```sql
-- Migration 004: Storyboard frames table and project style setting
CREATE TABLE IF NOT EXISTS storyboard_frames (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    shot_id UUID NOT NULL REFERENCES shots(id) ON DELETE CASCADE,
    file_path VARCHAR(1000) NOT NULL,
    thumbnail_path VARCHAR(1000),
    file_type VARCHAR(20) NOT NULL,
    is_selected BOOLEAN DEFAULT FALSE,
    generation_source VARCHAR(20) DEFAULT 'user',
    generation_style VARCHAR(30),
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ
);

CREATE INDEX IF NOT EXISTS ix_storyboard_frames_shot_id ON storyboard_frames(shot_id);
```

```python
class StoryboardFrame(Base):
    __tablename__ = "storyboard_frames"

    id = Column(UUID(as_uuid=True), primary_key=True, default=uuid.uuid4)
    shot_id = Column(UUID(as_uuid=True), ForeignKey("shots.id", ondelete="CASCADE"), nullable=False, index=True)
    file_path = Column(String(1000), nullable=False)
    thumbnail_path = Column(String(1000), nullable=True)
    file_type = Column(String(20), nullable=False)
    is_selected = Column(Boolean, default=False)
    generation_source = Column(String(20), default="user")
    generation_style = Column(String(30), nullable=True)
    created_at = Column(DateTime(timezone=True), server_default=func.now())
    updated_at = Column(DateTime(timezone=True), onupdate=func.now())

    shot = sa_relationship("Shot", back_populates="storyboard_frames")
```
IMPORTANT: Use `String(20)` and `String(30)`, NOT `Enum()` -- avoids SQLite test failures.

Helper functions:
- `_verify_project_ownership(db, project_id, user_id)` -- same pattern as shots.py: query Project by id + owner_id, raise 404 if not found
- `_verify_shot_in_project(db, shot_id, project_id)` -- query Shot by id + project_id, raise 404 "Shot not found in project" if not found

Endpoints:

a) `POST /{project_id}/shots/{shot_id}/frames` -- multipart upload
   - Params: `project_id: UUID`, `shot_id: UUID`, `file: UploadFile = File(...)`, `current_user`, `db`
   - Verify project ownership, verify shot in project
   - Validate file extension against ALLOWED_IMAGE_EXTENSIONS, determine file_type = "image"
   - Read file content, validate size <= MAX_FILE_SIZE
   - Generate safe filename: `f"{uuid4()}.{ext}"`
   - Save to `os.path.join(settings.MEDIA_DIR, str(project_id), "storyboard", safe_name)` (create dirs with os.makedirs)
   - Set `file_url = f"/media/{project_id}/storyboard/{safe_name}"`
   - Set `thumbnail_path = file_url` (per CONTEXT.md: skip thumbnail generation, copy file_path)
   - Create StoryboardFrame record with shot_id, file_path=file_url, thumbnail_path=file_url, file_type="image", is_selected=False, generation_source="user"
   - db.add, db.commit, db.refresh, return frame
   - On DB error: clean up file from disk, db.rollback, re-raise
   - Response: `schemas.StoryboardFrameResponse`, status 201

b) `GET /{project_id}/shots/{shot_id}/frames` -- list frames by shot
   - Verify project ownership, verify shot in project
   - Query StoryboardFrame where shot_id matches, order by created_at
   - Return list as `List[schemas.StoryboardFrameResponse]`

c) `PATCH /{project_id}/frames/{frame_id}` -- update is_selected
   - Verify project ownership
   - Query frame by id, join to shot to verify shot.project_id matches project_id. If not found, raise 404.
   - If body.is_selected is True:
     - First: `db.query(database.StoryboardFrame).filter(database.StoryboardFrame.shot_id == frame.shot_id).update({"is_selected": False})`
     - Then: `frame.is_selected = True`
   - If body.is_selected is False: `frame.is_selected = False`
   - db.commit(), db.refresh(frame), return frame
   - Request body: `schemas.StoryboardFrameUpdate`
   - Response: `schemas.StoryboardFrameResponse`

d) `DELETE /{project_id}/frames/{frame_id}` -- delete frame and file
   - Verify project ownership
   - Query frame by id, join to shot to verify shot.project_id matches project_id. If not found, raise 404.
   - Delete file from disk (best-effort): resolve file_path (strip "/media/" prefix, join with settings.MEDIA_DIR), os.remove if exists. Same for thumbnail_path if different from file_path.
   - db.delete(frame), db.commit()
   - Return 204
